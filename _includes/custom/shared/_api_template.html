{% include label.html content="get" %} Retrieve a List of Objects for a Given Site

{% highlight shell%} {% if site.project == "platform" %} GET /v1/user/sites/{SITE_ID}/ {% else %} GET /user/USER_ID/site/{SITE_ID}/ {% endif %} {% endhighlight %}
